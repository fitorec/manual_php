Cookies

Creación:

 1 <?
 2 setcookie('nombreCookie','valor cookie');
 3 #definiendo la fecha de expiración
 4 setcookie('nombreCookie','valor cookie',time()+3600);
 5 #definiendo la ruta
 6 setcookie('nombreCookie','valor cookie',time()+3600,'/ruta/');
 7 #definiendo el dominio
 8 setcookie('nombreCookie','valor cookie',time()+3600,'/ruta/','dominio.com');
 9 #definiendo el campo seguro
10 setcookie('nombreCookie','valor cookie',time()+3600,'/ruta/','dominio.com',true);
11 ?>

Acceso a las cookies.

1 <?
2     echo $_COOKIE['nombreCookie'];
3     print_r($_COOKIE);
4 ?>
